// programs/solcipher-hedera/scripts/deploy.js
async function main() {
  const SolCipherHedera = await ethers.getContractFactory("SolCipherHedera");
  const contract = await SolCipherHedera.deploy();
  await contract.deployed();
  console.log("SolCipherHedera deployed to:", contract.address);
}

main().catch((error) => {
  console.error(error);
  process.exitCode = 1;
});
